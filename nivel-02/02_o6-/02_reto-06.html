Casino magico de operadores

Abri la consola para ver los resultados.
const feNumber = 8;
const amorosaNumber = 7;
let victorias = 0;
//Calculate: feNumber % 2 (is it even?)
const feRemainder = feNumber % 2;
//Calculate: amorosaNumber % 2 (is it even?)
const amorosaRemainder = amorosaNumber % 2;
//Caculate: feNumber ** 2
const squaredFe = feNumber ** 2;
//Caculate: amorosaNumber ** 2
const squaredAmorosa = amorosaNumber ** 2;
//Increment victories with ++
victorias++;
//Show all calculations with console.log()
console.log(feRemainder === 0 ? 'Fe es par' : 'Fe es impar');
console.log(amorosaRemainder === 0 ? 'Amorosa es par' : 'Amorosa es impar');
console.log(`Fe ${feNumber} ** 2 = ${squaredFe}`);
console.log(`Amorosa ${amorosaNumber} ** 2 = ${squaredAmorosa}`);
console.log(`Fe gano ${victorias} ronda ${victorias > 1 ? 's' : ''}`);
